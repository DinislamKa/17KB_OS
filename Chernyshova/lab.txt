#!/bin/bash
select var in "Текущий пользователь" "Объем используемой памяти" "Объем дискового" "Запущенные процессы" "Процессы, запущенные текущим пользователем" "Ситемная дата и время" "Время запуска системы" "Выход" ;
do
echo $var;
case "$var" in
"Текущий пользователь") whoami;;
"Объем используемой памяти") free -m;;
"Объем дискового пространства") df --total;;
"Запущенные процессы") ps - e;;
"Процессы, запущенные текущим пользователем") n=` who|cut -d' ' -f1|sort|uniq `; ps -u $n;;
"Системная дата и время") date -u;;
"Время запуска системы") uptime;;
"Выход") exit;;
esac
done
